\begin{algorithmic}[1]

\Function {$Canexecute$} {r,p,$q_{teleport}$,State}    
 \State Input:P and M;
 \State Output:Number of teleportation;
 \If{$g_{i}$ and $g_{r}$ are consequent  ||  (r==p)}
      \State  {$run_{p}=1$};
       \State  {Return (1)};
   \EndIf              
 \State $q_{c} \leftarrow$ control of  $g_{p}$
 \State $q_{t} \leftarrow$ target of  $g_{p}$  
 \If {No gates between $g_{r}$ and $g_{p}$}
     \State  {$run_{p}=1$};
       \State  {Return (1)};
   \EndIf      
  
 \State k=Find(p); 
  \State $q_{c1} \leftarrow$ control of $g_{k}$
 \State $q_{t1} \leftarrow$ target of  $g_{k}$  
 
 \If {$q_{c1}==q_{teleport}$  And  $P_{q_{t1}}\neq P_{q_{teleport}}$ And $State=='\'t'\'$}
    \State $run_{k}=0;$  
    \State Return (0);  
 \EndIf
   \If {$g_{k}$ is one-qubit gate  And  $P_{k}  \neq P_{q_{teleport}}$}
         \State Return (0);
    \EndIf

    \If {$k \neq r$}
      \State $Return(Canexecute(r,k,q_{teleport},State))$
      \EndIf
\EndFunction
\end{algorithmic}
}
\end{algorithm}
