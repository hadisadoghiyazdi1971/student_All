\begin{algorithm}
{
\fontsize{9}{1}
\caption{The algorithm converts quantum circuit to bigraph}
%\label{fig4}
\begin{algorithmic}[1]

\Function {$Main$} {M}    
 \State Input:P and M
 \State Output:Number of teleportation
  \For {$i=1:m$}
      \If  {$g_{i}==1$}
      \State Continue;
      \EndIf
     \If{$g_{i}$ is local gate || $g_{i}$ is one-qubit gate}
       \State {$g_{i}=1$} 
     \EndIf
 \State $q_{c} \triangleright control of gate i-th$;
 \State $q_{t} \triangleright target of gate i-th$;
 \State $Scr=Part[q_{c}];$
 \State $Part[q_{c}]=Part[q_{t}]$; 
 \For {$j=i+1:m$}
  \State $r=Canexecute(i,j)$;
  \If {$r==1$}
     \State  $g_{i}=1$;
   \EndIf
  \EndFor
 \EndFor
\EndFunction
\end{algorithmic}
}
\end{algorithm}
